git clone https://github.com/Kevlargo/Homework-2-Binary-Representation-Floating-point-Math-and-Bit-Manipulation.git

cp Homework-2-Binary-Representation-Floating-point-Math-and-Bit-Manipulation/hw2_.cpp

gcc -c hw2_.cpp

g++ -o hw2_.cpp

./hw2 loop bound loop counter
